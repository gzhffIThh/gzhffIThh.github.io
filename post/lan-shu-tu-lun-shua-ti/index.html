PART 1.最短路

P1948 [USACO08JAN]电话线Telephone Lines

题目链接

解析

  1. 分层最短路
  2. 加一维最短路

code

  1. 分层最短路（Dijkstra）
#include <bits/stdc++.h>
using namespace std;
typedef pair<int, int> PII;
const int MAXN = 1001 * 1001;
const int MAXP = 10001 * 1001;
inline int readint() {
    int a = 0, f = 1;
    char c = getchar();
    while(c > '9' || c < '0') { if(c == '-') f = -1; c = getchar(); }
    while(c >= '0' && c <= '9') { a = a * 10 + c - '0'; c = getchar(); }
    return a * f;
}
int n, p, k;
int head[MAXN], nxt[MAXP], v[MAXP], w[MAXP], cnt;
int dis[MAXN], used[MAXN];
void add(int x, int y, int z)
{
    nxt[++cnt] = head[x];
    head[x] = cnt;
    v[cnt] = y;
    w[cnt] = z;
}
int Dijkstra()
{
    memset(dis, 0x3f, sizeof dis);
    priority_queue<PII, vector<PII >, greater<PII > > heap;
    heap.push({0, 1});
    dis[1] = 0;
    while(!heap.empty())
    {
        while(used[heap.top().second]) heap.pop();
        int u = heap.top().second; heap.pop();
        used[u] = 1;
        for (int i = head[u]; i; i = nxt[i])
        {
            int z = max(w[i], dis[u]);
            if(dis[v[i]] > z)
            {
                dis[v[i]] = z;
                heap.push({dis[v[i]], v[i]});
            }
        }
        if(u == (k + 1) * n) return dis[u];
    }
    if(dis[(k + 1) * n] == 1061109567)
        return -1;
    return dis[(k + 1) * n];
}
int main()
{
    n = readint(), p = readint(), k = readint();
    for (int i = 1; i <= p; i++)
    {
        int x, y, z;
        x = readint(), y = readint(), z = readint();
        add(x, y, z);
        add(y, x, z);
        for (int j = 1; j <= k; j++)//k+1层
        {
            add(n * (j - 1) + x, n * j + y, 0);
            add(n * (j - 1) + y, n * j + x, 0);
            add(n * j + x, n * j + y, z);
            add(n * j + y, n * j + x, z);
        }
    }
    for (int j = 1; j <= k; j++)
        add(n * j, n * (j + 1), 0);//防止有路径小于k条边
    printf("%d", Dijkstra());
    return 0;
}
